<html>
<head>
    <title>Hyperlinks</title>
</head>
<body>
    <h2>Click the links below to visit different websites:</h2>
   <ol>
    <!-- Opens the link in the same tab -->
    <li><a href="https://www.google.com">Google</a><br></li>

    <!-- Opens the link in a new tab or window -->
