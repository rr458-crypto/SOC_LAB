<script>
let isLoggedIn = false;
let cart = [];

// ---------- PAGE CONTROL ----------
function showPage(p) {
  if (!isLoggedIn && (p === 'catalog' || p === 'cart')) {
    alert("Please login first");
    p = 'login';
  }
  document.querySelectorAll('.page').forEach(x => x.style.display = 'none');
  document.getElementById(p).style.display = 'block';
  if (p === 'cart') loadCart();
}

// ---------- REGISTER ----------
function register() {
  let username = regUser.value.trim();
  let password = regPass.value.trim();

  // ---- CLIENT SIDE VALIDATION ----
   if (username === "" && password === "") {
    regMsg.innerText = "Username and Password are required";
    return;
  }
  if (username === "") {
    regMsg.innerText = "Username is required";
    return;
  }

  if (username.length < 3) {
    regMsg.innerText = "Username must be at least 3 characters";
    return;
  }

  if (password === "") {
    regMsg.innerText = "Password is required";
    return;
  }

  if (password.length < 6) {
    regMsg.innerText = "Password must be at least 6 characters";
    return;
  }
  // ---- END VALIDATION ----

  localStorage.setItem("user", username);
  localStorage.setItem("pass", password);

  regUser.value = "";
  regPass.value = "";
  regMsg.innerText = "Registered successfully. Please login.";

  showPage('login');
  
}

// ---------- LOGIN ----------
function login() {
 // ---- CLIENT SIDE VALIDATION ----
  if (logUser.value.trim() === "") {
    logMsg.innerText = "Username is required";
    return;
  }

  if (logPass.value.trim() === "") {
    logMsg.innerText = "Password is required";
    return;
  }
  // ---- END VALIDATION ----  
  if (
    logUser.value === localStorage.getItem("user") &&
    logPass.value === localStorage.getItem("pass")
    ) {
    isLoggedIn = true;

    navRegister.classList.add("disabled");
    navLogin.classList.add("disabled");
    navCatalog.classList.remove("disabled");
    navCart.classList.remove("disabled");
    navLogout.classList.remove('disabled');

    logUser.value = "";
    logPass.value = "";
    logMsg.innerText = "";

    showPage('catalog');
  } else {
    logMsg.innerText = "Invalid credentials";
  }
}

// ---------- LOGOUT ----------
function logout() {
  isLoggedIn = false;
  cart = [];

  navRegister.classList.remove("disabled");
  navLogin.classList.remove("disabled");
  navCatalog.classList.add("disabled");
  navCart.classList.add("disabled");
  navLogout.classList.add('disabled');


  alert("Logged out");
  showPage('login');
}

  d.className = "item";
  d.innerHTML = `
    <img src="${p.img}">
    <h4>${p.name}</h4>
    <p>Rs ${p.price}</p>
    <button onclick="addToCart(${p.id})">Add to Cart</button>
  `;
  productGrid.appendChild(d);
});

// ---------- CART ----------
function addToCart(id) {
  let item = cart.find(c => c.id === id);
  if (item) {
    item.qty++;
  } else {
    let p = products.find(p => p.id === id);
    cart.push({ id:p.id, name:p.name, price:p.price, qty:1 });
   alert(p.name + " added to cart");
  }
   
}

function loadCart() {
  cartList.innerHTML = "";
  let grandTotal = 0;

  cart.forEach((c, i) => {
    cartList.innerHTML += `
      <div class="cart-item">
        <span>${c.name}</span>
        <span>Rs ${c.price}</span>
        <div>
          <button class="qty-btn" onclick="changeQty(${i},-1)">-</button>
          ${c.qty}
          <button class="qty-btn" onclick="changeQty(${i},1)">+</button>
        </div>
        <button onclick="removeItem(${i})">X</button>
      </div>
    `;
    grandTotal += c.price * c.qty;
  });

  document.getElementById("totalAmount").innerText = grandTotal;
}

function changeQty(i, val) {
  cart[i].qty += val;
  if (cart[i].qty <= 0) cart.splice(i, 1);
  loadCart();
}

function removeItem(i) {
  cart.splice(i, 1);
  loadCart();
}
